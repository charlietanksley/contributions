    # File lib/contributions/string_utils.rb, line 94
def self.practically_empty?(arg)
  if arg.empty?
    return true
  elsif !arg.match %r\w/
    return true
  else
    return false
  end
end

def self.short_dates(hash)
  old_date = hash[:date]
  hash[:date] = old_date.match(%r(\d{4}-\d{2}-\d{2})/)[1]

  hash
end

def self.zip_to_hash(keys, values)
  value = Hash.new
  zipped = keys.zip values
  zipped.each do |pair|
    value[pair.first] = pair.last || ''
  end

  value
end
